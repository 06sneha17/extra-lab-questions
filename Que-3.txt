#include <stdio.h>
int main() {
int n, i, j, temp;
printf("Enter size of array: ");
scanf("%d", &n);
int arr[n];
printf("Enter elements: ");
for (i = 0; i < n; i++)
scanf("%d", &arr[i]);
// Rearranging negatives first
j = 0;
for(i = 0; i < n; i++) {
    if (arr[i] < 0) {
        temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
        j++;
    }
}
printf("Rearranged array: ");
for (i = 0; i < n; i++)
printf("%d" , arr[i]);
    return 0;
}
