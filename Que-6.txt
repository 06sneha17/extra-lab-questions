#include <stdio.h>
int main() {
int n, i, flag = 1;
    printf("Enter size of array: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter elements: ");
    for (i = 0; i < n; i++)
    scanf("%d", &arr[i]);
for (i = 0; i < n / 2; i++) {
    if (arr[i] ! = arr[n - i - 1]) {
        flag = 0;
        break;
    }
    }
if (flag)
printf("Array is Palindrome\n");
else
printf("Array is Not Palindrome\n");
return 0;
}
