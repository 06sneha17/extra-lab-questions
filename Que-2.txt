#include <stdio.h>
int main() {
int n;
printf("Enter size of array: ");
scanf("%d", &n);
int arr[n];
printf("Enter %d elements:\n ", n);
for (int i = 0; i < n; i++) {
    scanf("%d", &arr[i]);
}
printf("Leaders in the array:\n ");
for(int i = 0; i < n; i++) {
    int sum = 0;
    for(int j = i + 1; j < n; j++) {
    sum += arr[j];
}
if (arr[i] > sum) {
    printf("%d", arr[i]);
}
}
    return 0;
}
